UUID-scoped sessions
Your client UUID is issued once and embedded directly in the exchange connection path.
connect.js
// Connect to a live exchange feed
const uuid = "8a2b8f4c-0f12-44af-baf0-6a0a7db46b33";
const exchange = "XNYS"; // New York Stock Exchange

const ws = new WebSocket(
  `ws://api.newexperience.com/exchanges/${uuid}/${exchange}`
);

ws.onmessage = (event) => {
  const data = JSON.parse(event.data);

  if (data.event === "price") {
    console.log(`${data.exchange} → ${data.price}`);
    // { event: "price", exchange: "XNYS",
    //   price: 19847.32, timestamp: 1782307214 }
  }

  if (data.event === "market_status") {
    console.log(`${data.exchange} is ${data.status}`);
  }
};

ws.onopen = () => console.log("Connected ✓");
ws.onerror = (e) => console.error("Error:", e);
